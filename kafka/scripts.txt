Создать топик
/opt/kafka/bin/kafka-topics.sh \
  --bootstrap-server localhost:9092 \
  --create \
  --topic logs_topic \
  --partitions 1 \
  --replication-factor 1

/opt/kafka/bin/kafka-topics.sh   --bootstrap-server localhost:9092 --list
